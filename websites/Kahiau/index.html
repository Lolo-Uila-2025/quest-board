🧩 Project 1: Arduino Interactive Scene Game

// 🛠️ Setup Function
void setup() {
  // 🧠 Initialize serial monitor and buttons
  Serial.begin(9600);
  pinMode(3, INPUT_PULLUP);
  pinMode(4, INPUT_PULLUP);
  startScene(); // 🚀 Start the first scene
}

// 🔁 Main Loop
void loop() {
  // 🤖 All logic happens in scene functions!
}

void waitForPress(){
  // ⏸️ Pause until either button is pressed
  while (digitalRead(3) == HIGH && digitalRead(4) == HIGH){
    // ⌛ Waiting...
  }
}

void waitForRelease(){
  // ⏬ Wait until both buttons are released
  while(digitalRead(3) == LOW  || digitalRead(4) == LOW) {
    delay(10); // ⏳ Debounce
  }
}

// 🌾 Start Scene
void startScene() {
  Serial.println("🌄 Open Field:");
  Serial.println("1️⃣ Go to portals");
  Serial.println("2️⃣ Go to portals");
  waitForPress();

  if (digitalRead(3) == LOW) {
    waitForRelease();
    portals();
  } else if (digitalRead(4) == LOW) {
    waitForRelease();
    portals();
  }
}

// 🌀 Portal Selection
void portals() {
  Serial.println("🚪 Portals:");
  Serial.println("1️⃣ Green portal");
  Serial.println("2️⃣ Blue portal");
  waitForPress();

  if (digitalRead(3) == LOW) {
    waitForRelease();
    green();
  } else if (digitalRead(4) == LOW) {
    waitForRelease();
    blue();
  }
}

// 💚 Green Portal
void green() {
  Serial.println("💚 Green Portal:");
  Serial.println("1️⃣ Left side");
  Serial.println("2️⃣ Right side");
  waitForPress();

  if (digitalRead(3) == LOW) {
    waitForRelease();
    Serial.println("💀 Game Over — Restarting...");
    startScene();
  } else if (digitalRead(4) == LOW) {
    waitForRelease();
    Serial.println("🏆 YOU WIN — Restarting...");
    startScene();
  }
}

// 🔵 Blue Portal
void blue() {
  Serial.println("🔵 Blue Portal:");
  Serial.println("1️⃣ Go left");
  Serial.println("2️⃣ Go right");
  waitForPress();

  if (digitalRead(3) == LOW) {
    Serial.println("☠️ Game Over — YOU DIED");
    startScene();
  } else if (digitalRead(4) == LOW) {
    waitForRelease();
    Serial.println("🏆 YOU WIN");
    startScene();
  }
}
